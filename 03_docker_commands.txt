docker run -d -p 8080:80 --name mynginx nginx
------> -d is used to run the container in the background
------> -p is to assign the port number before ":" is for the local host and after ":" is for the prot inside docker
------> mynginx is the container name
------> nginx is the image name

docker rm mynginx
----> rm is used to remove the container form the docker

docker start <container_name/container_id>
-----> start is used to start the container which is not running using the container name or contianer id
 
docker stop mynginx
----> used to stop the running container


docker rm -f  
---> force remove container eg: docker rm -f mynginx

docker rmi 
---> Used to remove images, not containers. eg: docker rmi nginx


docker rmi -f 
---> Force remove image even if: A stopped container is using it eg: docker rmi -f nginx


docker run -it ubuntu /bin/bash
----> -it is to run the container in interactive mode when container is not running

docker exec -it <container_id> bash
----> exec is used to execute some task when container is running

docker logs 590fb635c3d9e122fea40b2c9ae7ff565acd606964df73b1f6ad93d91a72db78
-----> gives the logs about the container used for debugging
